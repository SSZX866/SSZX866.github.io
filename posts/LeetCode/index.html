class Solution:
    def isPalindrome(self, x: int) -> bool:
        if x < 0:
            return False
        else:
            x = str(x)
            i = len(x) - 1
            j = 0
            while(i > j):
                if x[i] != x[j]:
                    return False
                else:
                    i -= 1
                    j += 1
            return True

class Solution:
    def romanToInt(self, s: str) -> int:
        dic = {'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}

        i = 0
        ans = 0
        while i < len(s)-1:
            if dic[s[i]] >= dic[s[i+1]]:
                ans += dic[s[i]]
            else:
                ans -= dic[s[i]]
            i += 1

        return ans + dic[s[-1]]

class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        if not strs: return ""
        mins = min(strs)
        maxs = max(strs)
        for i in range(0, len(mins)+1, 1):
            if mins[:i] != maxs[:i]:
                return mins[:i-1]
        return mins
#max()和max()函数除了返回最大值和最小值还可返回列表中最大和最小的元素，按照ascII值排，举例abb， aba，abac，最大为abb，最小为aba

# Definition for singly-linked list.

#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        l = ListNode()
        s = l
        while l1 and l2:
            if l1.val > l2.val:
                l.next = l2
                l2 = l2.next
            else:
                l.next = l1
                l1 = l1.next
            l = l.next
        if l1:
            l.next = l1
        if l2:
            l.next = l2
        return s.next

class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        a = 0
        i = 0
        j = len(nums) - 1
        if set(nums) == {val}:
            return 0
        while i < j :
            while nums[i] != val and i < j:
                i += 1
                a += 1
            while nums[j] == val and i < j:
                j -= 1
            if i != j:
                nums[i] = nums[j]
                j -= 1
                i += 1
                a += 1
        if i == j and nums[i] != val:
            a += 1
        return a

class Solution:
    def strStr(self, haystack: str, needle: str) -> int:
        if needle == '':return 0
        i=0
        j=0
        while i <= (len(haystack) - len(needle)):
            while haystack[i] == needle[j] and j < (len(needle) - 1):
                i += 1
                j += 1
            if j == len(needle) - 1 and haystack[i] == needle[j]:
                return i - j
            i = i - j + 1
            j = 0
        return -1

class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        p = head
        while p:
            q = p
            while p.next and p.val == p.next.val:
                p = p.next
            if q != p:
                q.next = p.next
            p = p.next
        return head

        if p == None or q == None:
            return p == None and q == None
        return p.val == q.val and self.isSameTree(p.left, q.left) and self.isSameTree(p.right, q.right)

class Solution:
    def isSymmetric(self, root: TreeNode) -> bool:
        if root == None:
            return True
        else:
            return self.A(root.left,root.right)
    def A(self, left, right):
        if left == None or right == None:
            return left == None and right == None
        else:
            return self.A(left.left, right.right) and self.A(left.right, right.left) and right.val == left.val
